152void gp_bar::setValue(uint16_t value)
153{
154 if (_previousValue != value)
155 {
156 _previousValue = value;
157
158 // Set the value label
159 switch (_gaugeType)
160 {
161 case gauge_type::BATTERY_VOLTAGE:
162
163 if (batteryTimer.cycleTrigger(1000))
164 {
165 _ui_bar_set_property(_bar, _UI_BAR_PROPERTY_VALUE, map(value, _minValue, _maxValue, 0, 100));
166 _ui_label_set_property(_barValue, _UI_LABEL_PROPERTY_TEXT, String((float)value / 1000.0, 1).c_str());
167 }
168 break;
169
170 case gauge_type::MANIFOLD_PRESSURE:
171 case gauge_type::OIL_PRESSURE:
172 case gauge_type::FUEL_FLOW:
173 _ui_bar_set_property(_bar, _UI_BAR_PROPERTY_VALUE, map(value, _minValue, _maxValue, 0, 100));
174 _ui_label_set_property(_barValue, _UI_LABEL_PROPERTY_TEXT, String((float)value * 0.0333, 1).c_str());
175 break;
176
177 default:
178 _ui_bar_set_property(_bar, _UI_BAR_PROPERTY_VALUE, map(value, _minValue, _maxValue, 0, 100));
179 _ui_label_set_property(_barValue, _UI_LABEL_PROPERTY_TEXT, String(value).c_str());
180 break;
181 }
182
183 // Set the the high warning color change except for the fuel level gauge
184 if (value >= _warningValue && value < _alertValue && _gaugeType != gauge_type::FUEL_LEVEL)
185 {
186 lv_obj_set_style_bg_color(_bar, lv_palette_main(LV_PALETTE_ORANGE), LV_PART_INDICATOR | LV_STATE_DEFAULT);
187 }
188
189 // Set the the high alert color change except for the fuel level gauge
190 else if (value >= _alertValue && _gaugeType != gauge_type::FUEL_LEVEL)
191 {
192 lv_obj_set_style_bg_color(_bar, lv_palette_main(LV_PALETTE_RED), LV_PART_INDICATOR | LV_STATE_DEFAULT);
193 }
194
195 // Set the low warning alert color change except for the duty cycle and manifold pressure gauges
196 else if (value <= _lowWarningValue && value > _lowAlertValue && _gaugeType != gauge_type::INJ_DUTY && _gaugeType != gauge_type::MANIFOLD_PRESSURE)
197 {
198 switch (_gaugeType)
199 {
200 case gauge_type::AIR_TEMP:
201 case gauge_type::COOLANT_TEMP:
202 case gauge_type::OIL_TEMP:
203 lv_obj_set_style_bg_color(_bar, lv_palette_main(LV_PALETTE_BLUE), LV_PART_INDICATOR | LV_STATE_DEFAULT);
204 break;
205
206 default:
207 lv_obj_set_style_bg_color(_bar, lv_palette_main(LV_PALETTE_ORANGE), LV_PART_INDICATOR | LV_STATE_DEFAULT);
208 break;
209 }
210 }
211
212 // Set the low alert color change except for the duty cycle and manifold pressure gauges
213 else if (value <= _lowAlertValue && _gaugeType != gauge_type::INJ_DUTY && _gaugeType != gauge_type::MANIFOLD_PRESSURE)
214 {
215 switch (_gaugeType)
